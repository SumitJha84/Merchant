    st.markdown("*Simulate how changes in merchant behavior affect health score and Revenue at Risk.*")

    mid = st.selectbox("Select Merchant", df["merchant_id"].sort_values().tolist())
    row = df[df["merchant_id"] == mid].iloc[0]

    col_ctrl, col_res = st.columns([1, 1.5])

    with col_ctrl:
        st.markdown("#### Adjust Parameters")
        new_refund = st.slider("Refund Rate (%)", 0.0, 60.0,
                               float(row["refund_rate"]*100), 0.5) / 100
        new_cb     = st.slider("Chargeback Rate (%)", 0.0, 30.0,
                               float(row["chargeback_rate"]*100), 0.1) / 100
        gmv_change = st.slider("GMV Change (%)", -80, 50, 0, 5) / 100
        drift_adj  = st.slider("Behavioral Drift", 0.0, 1.0,
                               float(row.get("drift_composite", 0.2)), 0.01)

    # Recompute health score
    from sklearn.preprocessing import MinMaxScaler
    def recompute(ref, cb, drift, surv):
        s1 = surv
        s2 = max(0, 1 - ref / 0.60)
        s3 = max(0, 1 - cb  / 0.30)
        s4 = max(0, 1 - drift)
        return round((0.35*s1 + 0.25*s2 + 0.20*s3 + 0.20*s4) * 100, 1)

    orig_score = row["health_score"]
    new_surv   = max(0, float(row["survival_prob_4w"]) - gmv_change * 0.3 - (new_refund - row["refund_rate"]) * 2)
    new_score  = recompute(new_refund, new_cb, drift_adj, new_surv)
    new_gmv    = float(row.get("gmv_4w_avg", 100000)) * (1 + gmv_change)
    new_rar    = new_gmv * 4 * (1 - new_surv) * (1 + new_cb)
    delta      = new_score - orig_score

    with col_res:
        st.markdown("#### Simulation Result")
        c1, c2 = st.columns(2)
        with c1:
            st.metric("Original Score", f"{orig_score:.0f}", None)
            st.metric("New Score", f"{new_score:.0f}", f"{delta:+.1f}")
        with c2:
            st.metric("Original RaR", f"Rs {row['revenue_at_risk']:,.0f}")
            st.metric("New RaR", f"Rs {new_rar:,.0f}",
                      f"Rs {new_rar - row['revenue_at_risk']:+,.0f}")

        # Before/after bar
        fig = go.Figure(go.Bar(
            x=["Original", "Simulated"],
            y=[orig_score, new_score],
            marker_color=["#60a5fa", "#f87171" if new_score < 50 else "#34d399"],
            text=[f"{orig_score:.0f}", f"{new_score:.0f}"],
            textposition="outside",
        ))
        fig.update_layout(
            title="Health Score: Before vs After",
            template="plotly_dark", paper_bgcolor="#1e2940",
            plot_bgcolor="#162032", font_color="#e2e8f0",
            height=260, yaxis_range=[0, 110],
        )
        st.plotly_chart(fig, use_container_width=True)
